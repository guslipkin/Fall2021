bins <- seq(2.5,100,5)
tut$s1Bin <- cut(tut$s1, breaks = seq(0,100,5), labels = bins)
tut$s2Bin <- cut(tut$s2, breaks = seq(0,100,5), labels = bins)

q29 <- tut %>%
  group_by(s1Bin) %>%
  count(t) %>%
  mutate(percTreatedBin = round(n / sum(n) * 100, 2),
         percTreatedTotal = round(n / length(tut$t[tut$t == 1]) * 100, 2)) %>%
  filter(t == 1) %>%
  select(-t)
q29$s1Bin <- as.numeric(levels(q29$s1Bin)[(q29$s1Bin)])
q29
q29 %>%
  ggplot() +
  geom_point(aes(x = s1Bin, y = percTreatedTotal)) +
  geom_vline(xintercept = 60, color = "black") +
  scale_x_continuous(breaks = bins) +
  scale_y_binned(n.breaks = 10, limits = c(0,35))
